15 vec3 CreateDirectionVector(std::mt19937 &engine,
16 quat transform_rotation,
17 vec4 yaw_pitch_range,
18 vec2 radius_range)
19 {
20 std::uniform_real_distribution<float> yaw_distribution(yaw_pitch_range.x, yaw_pitch_range.z);
21 std::uniform_real_distribution<float> pitch_distribution(yaw_pitch_range.y, yaw_pitch_range.w);
22 std::uniform_real_distribution<float> radius_distribution(radius_range.x, radius_range.y);
23
24 float yaw = yaw_distribution(engine);
25 float pitch = pitch_distribution(engine);
26 float radius = radius_distribution(engine);
27
28 vec3 direction = radius * vec3(
29 std::cos(yaw) * std::cos(pitch),
30 std::sin(yaw) * std::cos(pitch),
31 std::sin(pitch)
32 );
33
34 return direction * transform_rotation;
35 }

141 {

144
145 last_tick_time_ = time_now;
146
147 for (auto entity : view)
148 {
149 auto &transform = view.get<components::TransformComponent>(entity);
150 auto &emitter = view.get<components::ParticleEmitter>(entity);
151 if (emitter.freeze)

153 for (auto &particle : emitter.particles)
154 {
155 particle.life_begin += delta_time;
156 }

158 }

161 {

163 }
164 else

167 emitter.particles_last_second_count = 0;
168 }
169
170 if (emitter.particles.size() > emitter.maximum_amount_of_particles)
171 {
172 emitter.particles.resize(emitter.maximum_amount_of_particles);
173 }
174
175 for (auto it = emitter.particles.begin(); it != emitter.particles.end();)
176 {
177 if (it->life_end < time_now)
178 {
179 it = emitter.particles.erase(it);
180 }
181 else
182 {
183 it++;
184 }
185 }
186
187 for (auto &particle : emitter.particles)
188 {
189 particle.position += particle.velocity * delta_time;
190 particle.velocity += particle.acceleration * delta_time;
191 particle.rotation += particle.rotation_speed * delta_time;
192 }
193 if (emitter.particles_last_second_count >= emitter.emit_rate)
194 {
195 continue;
196 }
197 int32_t amount_of_particles = static_cast<uint32_t>(emitter.emit_rate * emitter.last_second_count - emitter.particles_last_second_count);
198
199 amount_of_particles = int32_t(std::min((int64_t)amount_of_particles, (int64_t)(emitter.maximum_amount_of_particles - emitter.particles.size())));
200 if (amount_of_particles <= 0)
201 {
202 continue;
203 }
204 emitter.particles_last_second_count += amount_of_particles;
205
206 emitter.particles.reserve(emitter.particles.size() + amount_of_particles);
207
208 for (int32_t i = 0; i < amount_of_particles; i++)
209 {
210 emitter.particles.emplace_back();
211 auto &particle = emitter.particles.back();
212 particle.position = transform.position +
213 CreateDirectionVector(random_engine_,
214 transform.rotation,
215 emitter.position_yaw_pitch_range,
216 emitter.position_radius);
217 particle.velocity = CreateDirectionVector(random_engine_,
218 transform.rotation,
219 emitter.velocity_yaw_pitch_range,
220 emitter.velocity_radius);
221
222 particle.acceleration = emitter.particle_acceleration;
223
224 particle.color = emitter.base_diffuse_color +
225 random::RandomVector4(random_engine_, vec4(-1), vec4(1)) * emitter.diffuse_variation;
226 particle.begin_size = Randomize(random_engine_, emitter.begin_size_range);
227 particle.end_size = Randomize(random_engine_, emitter.end_size_range);
228 particle.life_begin = time_now;
229 particle.life_end = time_now + Randomize(random_engine_, emitter.particle_lifespan_range);
230 particle.rotation = Randomize(random_engine_, emitter.rotation_range);
231 particle.rotation_speed = Randomize(random_engine_, emitter.rotation_speed_range);
232 particle.thickness = Randomize(random_engine_, emitter.thickness_range);
233 }
