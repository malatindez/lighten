24 void processMesh(std::vector<Vertex>& vertices_dest,
25 std::vector<uint32_t>& indices_dest,
26 std::vector<ModelMesh>& meshes,
27 aiMesh* mesh,
28 aiMatrix4x4 const& transformation)
29 {
30 std::vector<Vertex> vertices;
31 std::vector<uint32_t> indices;
32 for (uint32_t i = 0; i < mesh->mNumVertices; i++)
33 {
34 math::vec3 position = math::vec3{ 0 };
35 math::vec2 tex_coord = math::vec2{ 0 };
36 math::vec3 normal = math::vec3{ 0 };
37 math::vec3 tangent = math::vec3{ 0 };
38 math::vec3 bitangent = math::vec3{ 0 };
39 if (mesh->mVertices != nullptr)
40 {
41 position = math::vec3{ mesh->mVertices[i].x, mesh->mVertices[i].y, mesh->mVertices[i].z };
42 }
43 if (mesh->mTextureCoords != nullptr && mesh->mTextureCoords[0] != nullptr)
44 {
45 tex_coord = math::vec2{ mesh->mTextureCoords[0][i].x, mesh->mTextureCoords[0][i].y };
46 }
47 if (mesh->mNormals != nullptr)
48 {
49 normal = math::vec3{ mesh->mNormals[i].x, mesh->mNormals[i].y, mesh->mNormals[i].z };
50 }
51 if (mesh->mTangents != nullptr)
52 {
53 tangent = math::vec3{ mesh->mTangents[i].x, mesh->mTangents[i].y, mesh->mTangents[i].z };
54 }
55 if (mesh->mBitangents != nullptr)
56 {
57 bitangent = math::vec3{ mesh->mBitangents[i].x, mesh->mBitangents[i].y, mesh->mBitangents[i].z };
58 }
59 vertices.emplace_back(Vertex{
60 .position = position,
61 .tex_coord = tex_coord,
62 .normal = normal,
63 .tangent = tangent,
64 .bitangent = bitangent });
65 }
66 for (uint32_t i = 0; i < mesh->mNumFaces; i++)
67 {
68 aiFace const& face = mesh->mFaces[i];
69 for (uint32_t j = 0; j < face.mNumIndices; j++)
70 {
71 indices.push_back(face.mIndices[j]);
72 }
73 }
74 vertices_dest.insert(vertices_dest.end(), vertices.begin(), vertices.end());
75 indices_dest.insert(indices_dest.end(), indices.begin(), indices.end());
76 ModelMesh mesh__ = SetupMesh(std::move(vertices), std::move(indices), mesh->mMaterialIndex, transformation, MeshRange{
77 // we are gonna fill offsets later
78 .vertex_offset = std::numeric_limits<uint32_t>::max(),
79 .index_offset = std::numeric_limits<uint32_t>::max(),
80 .vertex_count = mesh->mNumVertices,
81 .index_count = mesh->mNumFaces * 3,
82 .bounding_box = math::Box {
83 .min = math::vec3{mesh->mAABB.mMin.x, mesh->mAABB.mMin.y, mesh->mAABB.mMin.z},
84 .max = math::vec3{mesh->mAABB.mMax.x, mesh->mAABB.mMax.y, mesh->mAABB.mMax.z}
85 } }
86 );
87 meshes.push_back(std::move(mesh__));
88 }

128 std::optional<uint64_t> ModelLoader::Load(std::filesystem::path const& path)
129 {
130 if (auto it = instance_->models_.find(std::filesystem::hash_value(path));
131 it != instance_->models_.end())
132 {
133 return it->second;
134 }
135
136 Assimp::Importer importer;
137 aiScene const* scene_ptr = importer.ReadFile(path.string().c_str(),
138 aiProcess_Triangulate |
139 aiProcess_MakeLeftHanded |
140 aiProcess_FlipUVs |
141 aiProcess_FlipWindingOrder |
142 aiProcess_JoinIdenticalVertices |
143 (uint32_t)aiProcess_GenBoundingBoxes |
144 aiProcess_CalcTangentSpace |
145 aiProcess_OptimizeGraph |
146 aiProcess_OptimizeMeshes |
147 aiProcess_ImproveCacheLocality |
148 aiProcess_RemoveRedundantMaterials |
149 aiProcess_FindDegenerates |
150 aiProcess_FindInvalidData |
151 aiProcess_GenSmoothNormals |
152 aiProcess_SplitLargeMeshes |
153 aiProcess_SortByPType |
154 aiProcess_FindInstances |
155 aiProcess_ValidateDataStructure);
156
157 if (scene_ptr == nullptr)
158 {
159 utils::AlwaysAssert(false, "Failed to load model @ " + path.string());
160 return std::nullopt;
161 }
162 auto model_folder = std::filesystem::absolute(path.parent_path());
163
164 std::vector<Vertex> vertices;
165 std::vector<uint32_t> indices;
166 std::vector<ModelMesh> meshes;
167 std::vector<Material> materials;
168 ProcessMaterials(materials, scene_ptr, model_folder);
169 processNode(vertices, indices, meshes, scene_ptr->mRootNode, scene_ptr);
170 uint32_t index_offset = 0;
171 uint32_t vertex_offset = 0;
172 math::vec3 min{ std::numeric_limits<float>::max() };
173 math::vec3 max{ std::numeric_limits<float>::min() };
174
175 for (auto& mesh : meshes)
176 {
177 auto& mesh_range = mesh.mesh_range;
178 auto mesh_min = (math::vec4(mesh_range.bounding_box.min, 1) * mesh.mesh_to_model).xyz;
179 auto mesh_max = (math::vec4(mesh_range.bounding_box.max, 1) * mesh.mesh_to_model).xyz;
180 math::rmin(min, mesh_min);
181 math::rmin(min, mesh_max);
182 math::rmax(max, mesh_min);
183 math::rmax(max, mesh_max);
184
185 mesh_range.index_offset = index_offset;
186 mesh_range.vertex_offset = vertex_offset;
187 index_offset += mesh_range.index_count;
188 vertex_offset += mesh_range.vertex_count;
189 }
190 for (auto& mesh : meshes)
191 {
192 mesh.triangle_octree.initialize(mesh);
193 }
194
195 uint64_t rv = ModelSystem::instance().AddModel(Model{
196 .name = scene_ptr->mName.C_Str(),
197 .bounding_box = math::Box{.min = min, .max = max},
198 .meshes = std::move(meshes),
199 .materials = std::move(materials),
200 .vertices = direct3d::ImmutableVertexBuffer<Vertex>(vertices),
201 .indices = direct3d::ImmutableIndexBuffer<uint32_t>(indices) });
202 instance_->models_.emplace(std::pair<size_t, uint64_t>{std::filesystem::hash_value(path), rv});
203 return rv;
204 }

433 // create a unit sphere using
434 render::Mesh create_unit_sphere(uint32_t precision)
435 {
436 render::Mesh return_value;
437 return_value.vertices.resize(precision * precision);
438 return_value.indices.resize(precision * precision * 6);
439 for (uint32_t i = 0; i < precision; i++)
440 {
441 for (uint32_t j = 0; j < precision; j++)
442 {
443 float x = (float)i / (float)precision;
444 float y = (float)j / (float)precision;
445 float theta = x * 2.0f * core::math::numbers::pi_v<float>;
446 float phi = y * core::math::numbers::pi_v<float>;
447 float sin_theta = sin(theta);
448 float cos_theta = cos(theta);
449 float sin_phi = sin(phi);
450 float cos_phi = cos(phi);
