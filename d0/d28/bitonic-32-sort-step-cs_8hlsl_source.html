41[numthreads(256, 1, 1)]
42void BitonicSortStep(uint3 Gid: SV_GroupID,
43 uint3 GTid: SV_GroupThreadID)
44{
45 int4 tgp;
46
47 tgp.x = Gid.x * 256;
48 tgp.y = 0;
49 tgp.z = g_NumElements.x;
50 tgp.w = min(512, max(0, g_NumElements.x - Gid.x * 512));
51
52 uint localID = tgp.x + GTid.x; // calculate threadID within this sortable-array
53
54 uint index_low = localID & (job_params.x - 1);
55 uint index_high = 2 * (localID - index_low);
56
57 uint index = tgp.y + index_high + index_low;
58 uint nSwapElem = tgp.y + index_high + job_params.y + job_params.z * index_low;
59
60 if (nSwapElem < tgp.y + tgp.z)
61 {
62 float2 a = Data[index];
63 float2 b = Data[nSwapElem];
64
65 if (a.x > b.x)
66 {
67 Data[index] = b;
68 Data[nSwapElem] = a;
69 }
70 }
