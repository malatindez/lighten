40 mat4 CreateViewProjection(vec3 const &position, vec3 const &forward, vec3 const &up, float degrees = radians(90.0f))
41 {
42 static float kShadowRange = 100.0f;
43 mat4 rv{ 1 };
44 vec3 right = -normalize(cross(forward, up));
45 rv[0] = vec4{ right, 0 };
46 rv[1] = vec4{ up, 0 };
47 rv[2] = vec4{ forward, 0 };
48 rv[3] = vec4{ position, 1 };
49 return perspective(degrees, 1.0f, kShadowRange, 0.01f) * invert_orthonormal(rv);
50 }

52 mat4 CreateOrthoViewProjection(vec3 const &position, vec3 const &direction, float resolution)
53 {
54 vec3 up = vec3(0.0f, 1.0f, 0.0f);
55 vec3 right = normalize(cross(direction, up));
56 up = normalize(cross(right, direction));
57 mat4 view = look_at(position, position + direction, up);
58 mat4 projection = ortho(-resolution, resolution, -resolution, resolution, 10000.0f, 0.0f);
59 return projection * view;
60 }
