350 void GrassRenderSystem::OnInstancesUpdate(core::Scene *scene)
351 {
352 auto view = scene->registry.view<components::GrassComponent>();
353 for (auto entity : view)
354 {
355 auto &grass_component = view.get<components::GrassComponent>(entity);
356 if (materials_.size() <= grass_component.material_id)
357 {
358 continue;
359 }
360 auto &material_instance = *(materials_.begin() + grass_component.material_id);
361 if (std::find_if(material_instance.instances.begin(), material_instance.instances.end(),
362 [entity](auto const &instance) constexpr -> bool
363 {
364 return instance == entity;
365 }) != material_instance.instances.end())
366 {
367 continue;
368 }
369 material_instance.instances.push_back(entity);
370 }
371 if (view.empty())
372 {
373 return;
374 }
375 std::vector<GPUGrassInstance> grass_instances;
376 for (auto const &material : materials_)
377 {
378 for (auto entity : material.instances)
379 {
380 auto &grass_component = view.get<components::GrassComponent>(entity);
381 grass_instances.reserve(grass_instances.size() + grass_component.grass_instances_.size());
382 auto &atlas_data = material.material.atlas_data;
383 auto &atlas_size = material.material.atlas_size;
384 for (uint32_t current_atlas_id = 0; current_atlas_id < atlas_data.size(); current_atlas_id++)
385 {
386 auto begin = std::lower_bound(grass_component.grass_instances_.begin(), grass_component.grass_instances_.end(), current_atlas_id,
387 [](auto const &lhs, auto const &rhs) constexpr -> bool
388 {
389 return lhs.atlas_id < rhs;
390 });
391 auto end = std::upper_bound(begin, grass_component.grass_instances_.end(), current_atlas_id,
392 [](auto const &lhs, auto const &rhs) constexpr -> bool
393 {
394 return lhs < rhs.atlas_id;
395 });
396
397 core::math::uivec2 from = atlas_size * atlas_data[current_atlas_id].xy;
398 core::math::uivec2 to = atlas_size * atlas_data[current_atlas_id].zw;
399 for (auto &it = begin; it != end; ++it)
400 {
401 auto &instance = *it;
402 grass_instances.emplace_back(GPUGrassInstance{
403 .position = instance.position,
404 .size = instance.size,
405 .rotation = instance.rotation,
406 .from_uv_x = static_cast<uint16_t>(from.x),
407 .from_uv_y = static_cast<uint16_t>(from.y),
408 .to_uv_x = static_cast<uint16_t>(to.x),
409 .to_uv_y = static_cast<uint16_t>(to.y)});
410 }
411 }
412 }
413 }
414
415 if (grass_instances.size() <= 0)
416 {
417 return;
418 }
419 instance_buffer_.Init(std::span(grass_instances));
420 scene->renderer->light_render_system().ScheduleShadowMapUpdate();
421 }
