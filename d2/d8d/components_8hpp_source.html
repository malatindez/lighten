609 struct ForceField final : Component
610 {
612 enum class Type : uint8_t
613 {
614 // Point force field is a sphere that applies a force to all the particles inside it
615 // Basically works like this: applied_force += force * (1 - (distance / radius) ^ falloff)
620 Point = 0,
621 // Directional force field is a plane that applies a force to all the particles in front of it
622 // Basically works like this: applied_force += direction * force * (1 - (dot(normal, particle_position) / radius) ^ falloff)
623 // where direction = transform.forward
628 Directional = 1,
629 // OmniDirectional force field is a sphere field that applies force inwards of this sphere from all directions
630 // Can be used to simulate the air/water/whatever resistance
631 // You can calculate approximate force for the field like this:
632 // 0.5 * fluid_density * average_particle_velocity ^ 2 * average_particle_area * fluid_drag_coefficient
633 // It is pretty costy to simulate the whole fluid dynamics, so this is a good approximation
634 // Basically works like this: applied_force += -normalized(velocity.direction) * force
643 OmniDirectional = 2
644 };
