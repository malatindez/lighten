189 {
190 if (swapchain_.valid())
191 {
192 deinit();
193 }
194 DXGI_SWAP_CHAIN_DESC1 desc;
195 ZeroMemory(&desc, sizeof(DXGI_SWAP_CHAIN_DESC1));
196 desc.Width = window_size.x;
197 desc.Height = window_size.y;
198 desc.Format = DXGI_FORMAT_R8G8B8A8_UNORM;
199 desc.Stereo = FALSE;
200 desc.SampleDesc.Count = 1;
201 desc.SampleDesc.Quality = 0;
202 desc.BufferUsage = DXGI_USAGE_RENDER_TARGET_OUTPUT;
203 desc.BufferCount = 2;
204 desc.Scaling = DXGI_SCALING_NONE;
205 desc.SwapEffect = DXGI_SWAP_EFFECT_FLIP_DISCARD;
206 desc.AlphaMode = DXGI_ALPHA_MODE_UNSPECIFIED;
207 desc.Flags = 0;
208
209 IDXGISwapChain1 *swapchain = nullptr;
210 SetLastError(0);
211 AlwaysAssert(api().factory5->CreateSwapChainForHwnd(api().device, hWnd, &desc, nullptr, nullptr, &swapchain),
