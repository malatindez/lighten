125 bool SortLib::SortInitial(uint32_t max_size)
126 {
127 bool bDone = true;
128
129 // calculate how many threads we'll require:
130 // we'll sort 512 elements per CU (threadgroupsize 256)
131 // maybe need to optimize this or make it changeable during init
132 // TGS=256 is a good intermediate value
133
134 unsigned int numThreadGroups = ((max_size - 1) >> 9) + 1;
135 utils::Assert(numThreadGroups <= kMaxNumTG, "Too many thread groups: " + std::to_string(numThreadGroups) + "! Max is " + std::to_string(kMaxNumTG));
136 // To disable std::to_string on release builds
137 utils::AlwaysAssert(numThreadGroups <= kMaxNumTG);
138
139 if (numThreadGroups > 1)
140 bDone = false;
141
142 // sort all buffers of size 512 (and presort bigger ones)
143 lds_compute_shader_.Bind();
144 direct3d::api().devcon4->DispatchIndirect(indirect_sort_args_buffer_.buffer(), 0);

148 bool SortLib::SortIncremental(uint32_t presorted, uint32_t max_size)
149 {
150 bool bDone = true;
151 sort_step_compute_shader_.Bind();
152
153 // prepare thread group description data
154 unsigned int numThreadGroups = 0;
155
156 if (max_size > presorted)
157 {
158 if (max_size > presorted * 2)
159 bDone = false;
160
161 unsigned int pow2 = presorted;
162 while (pow2 < max_size)
163 pow2 *= 2;
164 numThreadGroups = pow2 >> 9;
165 }
166
167 unsigned int nMergeSize = presorted * 2;
168 for (unsigned int nMergeSubSize = nMergeSize >> 1; nMergeSubSize > 256; nMergeSubSize = nMergeSubSize >> 1)
169 // for( int nMergeSubSize=nMergeSize>>1; nMergeSubSize>0; nMergeSubSize=nMergeSubSize>>1 )
170 {
171 D3D11_MAPPED_SUBRESOURCE mapped_resource;
172 direct3d::api().devcon4->Map(pcb_dispatch_info_.buffer(), 0, D3D11_MAP_WRITE_DISCARD, 0, &mapped_resource);
173 core::math::ivec4 *sc = (core::math::ivec4 *)mapped_resource.pData;
174 sc->x = nMergeSubSize;
175 if (nMergeSubSize == nMergeSize >> 1)
176 {
177 sc->y = (2 * nMergeSubSize - 1);
178 sc->z = -1;
179 }
180 else

182 sc->y = nMergeSubSize;
183 sc->z = 1;
184 }
185 sc->w = 0;
186 direct3d::api().devcon4->Unmap(pcb_dispatch_info_.buffer(), 0);
187
188 direct3d::api().devcon4->Dispatch(numThreadGroups, 1, 1);
189 }
190
191 inner_compute_shader_.Bind();
192 direct3d::api().devcon4->Dispatch(numThreadGroups, 1, 1);
193
