33void sample_motion(float2 this_uv, float2 next_uv, float frameFracTime, out float2 uvA, out float2 uvB)
34{
35 float2 mvA = g_emva1.Sample(g_bilinear_clamp_sampler, this_uv).gb;
36 float2 mvB = g_emva1.Sample(g_bilinear_clamp_sampler, next_uv).gb;
37
38 // frameFracTime is in range [0, 1]
39
40 uvA = this_uv; // this frame UV
41 uvA -= mvA * g_mvScale * frameFracTime;
42
43 uvB = next_uv; // next frame UV
44 uvB -= mvB * g_mvScale * (frameFracTime - 1.f);
45}

54float3 ComputePointLightEnergy(PointLight point_light,
55 float3 posWS,
56 float3 rlt,
57 float3 botBF)

64 light_dir /= light_dist;

111float3 ComputeSpotLightEnergy(SpotLight spot_light, float3 posWS, float3 rlt, float3 botBF)
112{
113 float3 camera_right = g_inv_view[0].xyz;
114 float3 camera_up = g_inv_view[1].xyz;
115 float3 camera_forward = g_inv_view[2].xyz;
116 float3 light_dir = spot_light.position - posWS;
117 float light_dist = length(light_dir);
118 light_dir /= light_dist;
119
120 float3 cone_dir_normalized = normalize(spot_light.cone_direction);
121 float cosLightDir = dot(light_dir, cone_dir_normalized);
122 float cos_angle = dot(cone_dir_normalized, light_dir);
123 float cos_cutoff = cos(spot_light.inner_cutoff);
124 float cos_outer_cutoff = cos(spot_light.outer_cutoff);
125
126 if (cosLightDir > cos_outer_cutoff)
127 {
128 float sina;
129 float cosa;
130 float solid_angle = CalculateSolidAngle(light_dist, spot_light.radius, sina, cosa);
131
132 float cosRight = dot(camera_right, light_dir);
133 float cosUp = dot(camera_up, light_dir);
134 float cosForward = dot(camera_forward, light_dir);
135
136 float multiplier = 0.0f;
137 multiplier += abs(cosRight) * (cosRight > 0 ? rlt.r : rlt.g);
138 multiplier += abs(cosUp) * (cosUp > 0 ? rlt.b : botBF.r);
139 multiplier += abs(cosForward) * (cosForward > 0 ? botBF.g : botBF.b);
140
141 float attenuation = saturate((cos_angle - cos_outer_cutoff) / (cos_cutoff - cos_outer_cutoff));
142 attenuation *= attenuation;
143
144 return spot_light.color * solid_angle * multiplier * attenuation;
145 }
146 return float3(0, 0, 0);
147}
