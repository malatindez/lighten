7 /*
8 Converts a 128-bits unsigned int to an UUIDv4 string representation.
9 Uses SIMD via Intel's AVX2 instruction set.
10 */
11 void inline m128itos(__m128i x, char *mem) {
12 // Expand each byte in x to two bytes in res
13 // i.e. 0x12345678 -> 0x0102030405060708
14 // Then translate each byte to its hex ascii representation
15 // i.e. 0x0102030405060708 -> 0x3132333435363738
16 const __m256i mask = _mm256_set1_epi8(0x0F);
17 const __m256i add = _mm256_set1_epi8(0x06);
18 const __m256i alpha_mask = _mm256_set1_epi8(0x10);
19 const __m256i alpha_offset = _mm256_set1_epi8(0x57);
20
21 __m256i a = _mm256_castsi128_si256(x);
22 __m256i as = _mm256_srli_epi64(a, 4);
23 __m256i lo = _mm256_unpacklo_epi8(as, a);
24 __m128i hi = _mm256_castsi256_si128(_mm256_unpackhi_epi8(as, a));
25 __m256i c = _mm256_inserti128_si256(lo, hi, 1);
26 __m256i d = _mm256_and_si256(c, mask);
27 __m256i alpha = _mm256_slli_epi64(_mm256_and_si256(_mm256_add_epi8(d, add), alpha_mask), 3);
28 __m256i offset = _mm256_blendv_epi8(_mm256_slli_epi64(add, 3), alpha_offset, alpha);
29 __m256i res = _mm256_add_epi8(d, offset);
30
31 // Add dashes between blocks as specified in RFC-4122
32 // 8-4-4-4-12
33 const __m256i dash_shuffle = _mm256_set_epi32(0x0b0a0908, 0x07060504, 0x80030201, 0x00808080, 0x0d0c800b, 0x0a090880, 0x07060504, 0x03020100);
34 const __m256i dash = _mm256_set_epi64x(0x0000000000000000ull, 0x2d000000002d0000ull, 0x00002d000000002d, 0x0000000000000000ull);
35
36 __m256i resd = _mm256_shuffle_epi8(res, dash_shuffle);
37 resd = _mm256_or_si256(resd, dash);
38
39 _mm256_storeu_si256((__m256i *)mem, endianness::betole256(resd));
40 *(uint16_t *)(mem + 16) = endianness::betole16(_mm256_extract_epi16(res, 7));
41 *(uint32_t *)(mem + 32) = endianness::betole32(_mm256_extract_epi32(res, 7));
42 }

44 /*
45 Converts an UUIDv4 string representation to a 128-bits unsigned int.
46 Uses SIMD via Intel's AVX2 instruction set.
47 */
48 __m128i inline stom128i(const char *mem) {
49 // Remove dashes and pack hex ascii bytes in a 256-bits int
50 const __m256i dash_shuffle = _mm256_set_epi32(0x80808080, 0x0f0e0d0c, 0x0b0a0908, 0x06050403, 0x80800f0e, 0x0c0b0a09, 0x07060504, 0x03020100);
51
52 __m256i x = endianness::betole256(_mm256_loadu_si256((__m256i *)mem));
53 x = _mm256_shuffle_epi8(x, dash_shuffle);
54 x = _mm256_insert_epi16(x, endianness::betole16(*(uint16_t *)(mem + 16)), 7);
55 x = _mm256_insert_epi32(x, endianness::betole32(*(uint32_t *)(mem + 32)), 7);
56
57 // Build a mask to apply a different offset to alphas and digits
58 const __m256i sub = _mm256_set1_epi8(0x2F);
59 const __m256i mask = _mm256_set1_epi8(0x20);
60 const __m256i alpha_offset = _mm256_set1_epi8(0x28);
61 const __m256i digits_offset = _mm256_set1_epi8(0x01);
62 const __m256i unweave = _mm256_set_epi32(0x0f0d0b09, 0x0e0c0a08, 0x07050301, 0x06040200, 0x0f0d0b09, 0x0e0c0a08, 0x07050301, 0x06040200);
63 const __m256i shift = _mm256_set_epi32(0x00000000, 0x00000004, 0x00000000, 0x00000004, 0x00000000, 0x00000004, 0x00000000, 0x00000004);
64
65 // Translate ascii bytes to their value
66 // i.e. 0x3132333435363738 -> 0x0102030405060708
67 // Shift hi-digits
68 // i.e. 0x0102030405060708 -> 0x1002300450067008
69 // Horizontal add
70 // i.e. 0x1002300450067008 -> 0x12345678
71 __m256i a = _mm256_sub_epi8(x, sub);
72 __m256i alpha = _mm256_slli_epi64(_mm256_and_si256(a, mask), 2);
73 __m256i sub_mask = _mm256_blendv_epi8(digits_offset, alpha_offset, alpha);
74 a = _mm256_sub_epi8(a, sub_mask);
75 a = _mm256_shuffle_epi8(a, unweave);
76 a = _mm256_sllv_epi32(a, shift);
77 a = _mm256_hadd_epi32(a, _mm256_setzero_si256());
78 a = _mm256_permute4x64_epi64(a, 0b00001000);
79
80 return _mm256_castsi256_si128(a);
81 }

243 /* Generates a new UUID */
244 UUID getUUID() {
245 // The two masks set the uuid version (4) and variant (1)
246 const __m128i and_mask = _mm_set_epi64x(0xFFFFFFFFFFFFFF3Full, 0xFF0FFFFFFFFFFFFFull);
247 const __m128i or_mask = _mm_set_epi64x(0x0000000000000080ull, 0x0040000000000000ull);
248 __m128i n = _mm_set_epi64x(distribution(*generator), distribution(*generator));
249 __m128i uuid = _mm_or_si128(_mm_and_si128(n, and_mask), or_mask);
250
251 return UUID(uuid);
252 }
253
254 private:
255 std::shared_ptr<RNG> generator;
256 std::uniform_int_distribution<uint64_t> distribution;
257 };
